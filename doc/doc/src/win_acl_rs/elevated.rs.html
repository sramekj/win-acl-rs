25/// Enables *SeSecurityPrivilege* privilege.
26/// This typically needs a process running as an Administrator.
27fn enable_se_security_privilege() -> Result<(), WinError> {
28    unsafe {
29        let mut token: HANDLE = null_mut();
30
31        winapi_bool_call!(OpenProcessToken(
32            GetCurrentProcess(),
33            TOKEN_ADJUST_PRIVILEGES | TOKEN_QUERY,
34            &mut token,
35        ));
36
37        let mut luid = LUID {
38            LowPart: 0,
39            HighPart: 0,
40        };
41        winapi_bool_call!(LookupPrivilegeValueW(ptr::null(), SE_SECURITY_NAME, &mut luid), {
42            CloseHandle(token);
43        });
44
45        let tp = TOKEN_PRIVILEGES {
46            PrivilegeCount: 1,
47            Privileges: [windows_sys::Win32::Security::LUID_AND_ATTRIBUTES {
48                Luid: luid,
49                Attributes: SE_PRIVILEGE_ENABLED,
50            }],
51        };
52        winapi_bool_call!(
53            AdjustTokenPrivileges(token, 0, &tp as *const _ as _, 0, null_mut(), null_mut()),
54            {
55                CloseHandle(token);
56            }
57        );
58
59        let err = GetLastError();
60        CloseHandle(token);
61
62        if err != ERROR_SUCCESS {
63            return Err(err.into());
64        }
65    }
66    Ok(())
67}

69/// Checks if the current process is running as an Administrator.
70pub fn is_admin() -> Result<bool, WinError> {
71    unsafe {
72        let mut token_handle = null_mut();
73        let mut token_elevation = TOKEN_ELEVATION { TokenIsElevated: 0 };
74
75        winapi_bool_call!(OpenProcessToken(GetCurrentProcess(), TOKEN_QUERY, &mut token_handle));
76
77        let mut size = 0;
78
79        winapi_bool_call!(
80            GetTokenInformation(
81                token_handle,
82                TokenElevation,
83                &mut token_elevation as *mut _ as *mut std::ffi::c_void,
84                size_of::<TOKEN_ELEVATION>() as u32,
85                &mut size,
86            ),
87            {
88                CloseHandle(token_handle);
89            }
90        );
91
92        CloseHandle(token_handle);
93        Ok(token_elevation.TokenIsElevated != 0)
94    }
95}

97pub trait PrivilegeLevel {}
98#[derive(Debug)]
99pub struct Unprivileged;
100#[derive(Debug)]
101pub struct Elevated;
102impl PrivilegeLevel for Unprivileged {}
103impl PrivilegeLevel for Elevated {}
104
105/// Represents the current privilege context.
106pub type PrivilegeToken = PrivilegeTokenImpl<Unprivileged>;
107
108#[derive(Debug)]
109pub struct PrivilegeTokenImpl<P: PrivilegeLevel> {
110    _marker: PhantomData<P>,
111}
112
113impl PrivilegeTokenImpl<Unprivileged> {
114    pub fn new() -> Self {
115        Self { _marker: PhantomData }
116    }
117
118    /// Try to elevate the current process.
119    pub fn try_elevate(self) -> Result<PrivilegeTokenImpl<Elevated>, WinError> {
120        enable_se_security_privilege()?;
121        Ok(PrivilegeTokenImpl { _marker: PhantomData })
122    }
123}
124
125impl Default for PrivilegeTokenImpl<Unprivileged> {
126    fn default() -> Self {
127        Self::new()
128    }
129}
130
131impl PrivilegeTokenImpl<Elevated> {
132    /// Drop back to unprivileged state.
133    pub fn drop_privileges(self) -> PrivilegeTokenImpl<Unprivileged> {
134        PrivilegeTokenImpl::new()
135    }
136}

138impl SecurityDescriptorImpl<Elevated> {
139    /// Creates a SecurityDescriptor from path to the "file object"
140    ///
141    /// # Arguments
142    ///
143    /// * `_token` - an elevated privilege token.
144    /// * `path` - Path to the file.
145    ///
146    /// # Returns
147    ///
148    /// A `SecurityDescriptor` on success.
149    pub fn from_path<P>(_token: &PrivilegeTokenImpl<Elevated>, path: P) -> Result<Self, WinError>
150    where
151        P: AsRef<Path>,
152    {
153        let wide_path = WideCString::new(OsStr::new(path.as_ref()));
154
155        Self::create_sd(
156            wide_path.as_ptr(),
157            SE_FILE_OBJECT,
158            OBJECT_SECURITY_INFORMATION::get_all(),
159        )
160    }
161
162    /// Creates a SecurityDescriptor from object name and object type.
163    ///
164    /// # Arguments
165    ///
166    /// * `_token` - an elevated privilege token.
167    /// * `handle` - name of the object. This could be many things (path to the file or directory, to network share, name of the printer, registry key, ...)
168    /// * `object_type` - a type of the object
169    ///
170    /// see [MSDN](https://learn.microsoft.com/en-us/windows/win32/api/accctrl/ne-accctrl-se_object_type)
171    ///
172    /// # Returns
173    ///
174    /// A `SecurityDescriptor` on success.
175    pub fn from_handle<S>(
176        _token: &PrivilegeTokenImpl<Elevated>,
177        handle: S,
178        object_type: SE_OBJECT_TYPE,
179    ) -> Result<Self, WinError>
180    where
181        S: AsRef<str>,
182    {
183        let wide_string = WideCString::new(handle.as_ref());
184        Self::create_sd(
185            wide_string.as_ptr(),
186            object_type,
187            OBJECT_SECURITY_INFORMATION::get_all(),
188        )
189    }
190}

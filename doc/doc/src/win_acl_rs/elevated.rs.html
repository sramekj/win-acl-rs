12pub fn enable_se_security_privilege() -> Result<(), WinError> {
13    unsafe {
14        let mut token: HANDLE = null_mut();
15        if OpenProcessToken(
16            GetCurrentProcess(),
17            TOKEN_ADJUST_PRIVILEGES | TOKEN_QUERY,
18            &mut token,
19        ) == 0
20        {
21            return Err(GetLastError().into());
22        }
23
24        let mut luid = LUID {
25            LowPart: 0,
26            HighPart: 0,
27        };
28        if LookupPrivilegeValueW(ptr::null(), SE_SECURITY_NAME, &mut luid) == 0 {
29            CloseHandle(token);
30            return Err(GetLastError().into());
31        }
32
33        let tp = TOKEN_PRIVILEGES {
34            PrivilegeCount: 1,
35            Privileges: [windows_sys::Win32::Security::LUID_AND_ATTRIBUTES {
36                Luid: luid,
37                Attributes: SE_PRIVILEGE_ENABLED,
38            }],
39        };
40
41        if AdjustTokenPrivileges(token, 0, &tp as *const _ as _, 0, null_mut(), null_mut()) == 0 {
42            CloseHandle(token);
43            return Err(GetLastError().into());
44        }
45
46        let err = GetLastError();
47        CloseHandle(token);
48
49        if err != ERROR_SUCCESS {
50            return Err(err.into());
51        }
52    }
53    Ok(())
54}

56pub fn is_admin() -> Result<bool, WinError> {
57    unsafe {
58        let mut token_handle = null_mut();
59        let mut token_elevation = TOKEN_ELEVATION { TokenIsElevated: 0 };
60
61        if OpenProcessToken(GetCurrentProcess(), TOKEN_QUERY, &mut token_handle) == 0 {
62            return Err(GetLastError().into());
63        }
64
65        let mut size = 0;
66        if GetTokenInformation(
67            token_handle,
68            TokenElevation,
69            &mut token_elevation as *mut _ as *mut std::ffi::c_void,
70            size_of::<TOKEN_ELEVATION>() as u32,
71            &mut size,
72        ) == 0
73        {
74            CloseHandle(token_handle);
75            return Err(GetLastError().into());
76        }
77
78        CloseHandle(token_handle);
79        Ok(token_elevation.TokenIsElevated != 0)
80    }
81}

86    use crate::utils::WideCString;

88    use std::path::Path;
89    use windows_sys::Win32::Security::Authorization::SE_FILE_OBJECT;
90    use windows_sys::Win32::Security::{
91        DACL_SECURITY_INFORMATION, GROUP_SECURITY_INFORMATION, OWNER_SECURITY_INFORMATION,
92        SACL_SECURITY_INFORMATION,
93    };
94
95    impl SecurityDescriptor {
96        // the process requires "SE_SECURITY_NAME" (SeSecurityPrivilege) privilege, would otherwise return WIN32_ERROR(1314) => "A required privilege is not held by the client"
97        // you can run "whoami /priv" to check it. You typically need to run the process as an Administrator and enable it using enable_se_security_privilege().
98        pub fn from_path_elevated<P>(path: P) -> Result<Self, WinError>
99        where
100            P: AsRef<Path>,
101        {
102            let wide_path = WideCString::new(OsStr::new(path.as_ref()));
103
104            SecurityDescriptor::create_sd(
105                wide_path.as_ptr(),
106                SE_FILE_OBJECT,
107                DACL_SECURITY_INFORMATION
108                    | GROUP_SECURITY_INFORMATION
109                    | OWNER_SECURITY_INFORMATION
110                    | SACL_SECURITY_INFORMATION,
111            )
112        }
113    }
